CREATE VIEW EMP_NEWYORK1 
AS SELECT * FROM EMPLOYEE 
WHERE LOCATION = 'NEW YOURK' 
SELECT * FROM EMP_NEWYORK1 

CREATE VIEW EMP_HALF 
AS SELECT EMP_NO, EMP_NAME, JOB_ID FROM EMPLOYEE 
SELECT * FROM EMP_HALF 

CREATE VIEW EMP_Q3 
AS SELECT * FROM EMPLOYEE 
WHERE LOCATION = 'NEW YOURK' AND EMP_SAL > 3000 
SELECT * FROM EMP_Q3 

CREATE VIEW EMP_Q4 
AS SELECT COUNT(EMP_NO) AS "NUM_OF_EMP" FROM EMPLOYEE 
SELECT * FROM EMP_Q4
