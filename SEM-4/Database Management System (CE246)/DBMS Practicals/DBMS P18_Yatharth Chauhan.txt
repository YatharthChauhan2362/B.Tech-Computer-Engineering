CREATE OR REPLACE FUNCTION SALARY(S1 NUMBER) 
RETURN NUMBER IS 
SAL NUMBER; 

BEGIN 
SAL := S1; 
SAL := SAL +(SAL*0.1); 
RETURN SAL; 
END; 
 
DECLARE 
 
E_NO NUMBER :=:ENTER_NO; 
 
A NUMBER; 
 
B NUMBER; 
 
BEGIN 
   SELECT EMP_SAL INTO A FROM EMPLOYEE WHERE EMP_NO = E_NO; 
 
   B := SALARY(A); 
 
   DBMS_OUTPUT.PUT_LINE('UPDATED SALARY IS: ' || B); 
 
   UPDATE EMPLOYEE SET EMP_SAL = B WHERE EMP_NO = E_NO; 
END; 
 
SELECT * FROM EMPLOYEE;
