DECLARE 
 
D_NO NUMBER :=:ENTER_DEPT_NO; 
 
CURSOR C1 IS 
 
SELECT * FROM EMPLOYEE WHERE DEPT_NO = D_NO; 
 
EMP_RECORD C1%ROWTYPE; 
 
BEGIN 
 
OPEN C1; 
 
LOOP 
 
  FETCH C1 INTO EMP_RECORD; 
 
  EXIT WHEN C1%NOTFOUND; 
 
  IF(EMP_RECORD.EMP_SAL < 1000 AND (EMP_RECORD.MANAGER_ID = 105 OR EMP_RECORD.MANAGER_ID = 107)) THEN 
 
  DBMS_OUTPUT.PUT_LINE(EMP_RECORD.L_NAME || 'DUE FOR A RAISE ');  
 
  ELSE 
 
   DBMS_OUTPUT.PUT_LINE(EMP_RECORD.L_NAME || 'NOT DUE FOR A RAISE 
');  
  
   END IF; 
 
END LOOP; 
CLOSE C1; 
 
END
