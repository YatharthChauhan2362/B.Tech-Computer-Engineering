CREATE TABLE EMP_NEW AS(SELECT * FROM EMPLOYEE) 
SELECT * FROM EMP_NEW; 
CREATE OR REPLACE TRIGGER SALARY 
BEFORE INSERT OR UPDATE OR DELETE ON EMP_NEW 
FOR EACH ROW 
BEGIN 
   
CASE 
   
WHEN UPDATING THEN 
 
   DBMS_OUTPUT.PUT_LINE('UPDATING');     
   DBMS_OUTPUT.PUT_LINE('NEW SAL'||:NEW.EMP_SAL); 
 
WHEN INSERTING THEN 
    
   DBMS_OUTPUT.PUT_LINE('INSERTING'); 
 
WHEN DELETING THEN 
    
   DBMS_OUTPUT.PUT_LINE('DELETING'); 
 
 
END CASE; 
 
  END; 
 
UPDATE EMP_NEW SET EMP_SAL=2000 WHERE DEPT_NO=10;
